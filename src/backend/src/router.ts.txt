import * as express from 'express';
import { 
  getPlayer,
  getAllPlayers,
  createPlayer,
  updatePlayer,
  deletePlayer,
} from './controllers/PlayerController';

import {
  getAward,
  createAward,
  updateAward,

} from './controllers/AwardController';

import {
  getPartie,
  createPartie,
  updatePartie,
  deletePartie,
} from './controllers/PartieController';

import {
 
  getChampionnat,
  createChampionnat,
  getCurrentSeasonTotalScores,
  updateChampionnatForNewSeason,
  deleteChampionnat,
} from './controllers/ChampionnatController';

const router = express.Router();

// Player routes
router.get('/players', getPlayer);
router.get('/allplayers', getAllPlayers);
router.post('/players', createPlayer);
router.put('/players/:id', updatePlayer);
router.delete('/players/:id', deletePlayer);

// Award routes
router.get('/awards', getAward);
router.post('/awards', createAward);
router.put('/awards/:id', updateAward);
// router.delete('/awards/:id', deleteAward);

// Partie routes
router.get('/parties', getPartie);
router.post('/parties', createPartie);
router.put('/parties/:id', updatePartie);
router.delete('/parties/:id', deletePartie);

// Championnat routes
router.get('/championnats', getChampionnat);
router.post('/championnats', createChampionnat);
router.put('/championnats/:id', updateChampionnatForNewSeason);
router.delete('/championnats/:id', deleteChampionnat);
router.get('/api/championnats/current-season-total-scores', getCurrentSeasonTotalScores);

export default router;
